193 double ActSTPeripheral::calcxnSigma(ActNucleiData* data) {
194 
195  if (data == 0) {return 0.0;}
196 
197  // (p, xn) ST'77
198  // sig = sige0*G(e)*H(zt,x)*Omega
199  double at = data->getat();
200  double zt = data->getzt();
201  double a = data->geta();
202  double z = data->getz();
203 
204  int ia = data->getia();
205  int iz = data->getiz();
206  int izt = data->getizt();
207 
208  double e = data->gete();
209  double x = data->getx();
210 
211  double e0pxn = 300.0*x + 500.0;
212  double nzt(0.0), comb(0.0);
213  if (fabs(zt) >= 1e-30) {
214  nzt = (at - zt)/zt;
215  comb = ((1.0 - x)/zt) + nzt;
216  }
217  double comb2 = _formulae.power((1.5 - comb), 5.0);
218  double dummysig = 0.5*_formulae.expfun(-90.0*comb2);
219 
220  // sige0 first
221  double sige0(0.0);
222  nzt = fmin(1.5, nzt);
223 
224  if (nzt <= 1.5 && nzt >= 1.28) {
225  double upt1 = (1.5 - nzt)/0.22;
226  double upt2 = (nzt - 1.28)/0.22;
227  double pxnsig1 = this->pxnsig(x, 1.28);
228  double pxnsig2 = this->pxnsig(x, 1.5);
229  sige0 = _formulae.power(pxnsig1, upt1)*_formulae.power(pxnsig2, upt2);
230  } else {
231  sige0 = this->pxnsig(x, nzt);
232  }
233  if (comb <= 1.2) {sige0 = fmin(dummysig, sige0);}
234 
235  // Energy dependence now
236  double bigg(0.0);
237  if (e >= e0pxn) {
238  bigg = this->pxng(at, e0pxn, x);
239  } else {
240  double df = this->pxnf(at, zt, a, z, e, x);
241  double dg = this->pxng(at, e, x);
242  double dh = this->pxnh(at, zt, a, z, e, x);
243  if (_debug == 1) {cout<<"df,dg,dh = "<<df<<" "<<dg<<" "<<dh<<endl;}
244  bigg = df*(dg + dh);
245  }
246 
247  // bigh factor now and special omega
248  double bigh(1.0);
249  if (izt >= 83) {
250  // For heavy fissionable nuclei
251  double pow1 = (zt - 80.0)/12.0;
252  double pow2 = _formulae.power(pow1, 5.0);
253  bigh = _formulae.expfun(0.4 - x*pow2);
254  bigh = fmin(1.0, bigh);
255  }
256  double omega(1.0);
257  if (ia == 9 && iz == 4) {
258  omega = 0.65;
259  } else if (ia == 13 && iz == 7) {
260  omega = 0.4;
261  }
262 
263  double sigma = sige0*bigg*bigh*omega;
264 
265  if (_debug == 1) {
266  cout<<"sig1 = "<<_sigma<<endl;
267  cout<<"values are "<<sige0<<" "<<bigg<<" "<<bigh<<" "<<omega<<endl;
268  }
269 
270  return sigma;
271

345

374  }

544  // x>=0 and at<35; use 33a,b for p,3pxn x>=0 for heavier tgts.
545  // Use crude estimate of last lines p.359 for large y reactions.
546  // First do lightest nuclei (eqn. 36). Note that xmax for
547  // these is set to 0.0 in ActSTSelect, so only (p,3p)
548  // should be involved.
549

551  // values of ichg accordingly so that we can obtain the 3pxn xsection.
552  // Scaling factors for the 4p and 5p reactions are applied later on.

556  ichg -= 10;
557  ihiy = 1;
558  } else if (ichg >= 50 && ichg <= 59) {
559  ichg -= 20;

569  sigma = 0.0022*a*a*sigmas; // 3pxn sigma

573  } else {

578  if (izt <= 7 && ix == 0) {sigma *= 1.5;}

586  if (iat > 70 && iat <= 209) {xmax = at*0.04 + 0.5;}
587  int ixmax = (int) xmax;

590  gotSigma = true;

608  hofe = fmin((e/200.0), 1.0);

622  hofe = fmin((e/200.0), 1.0);

625  double eratio(0.0);

627  eratio = e/denom;
628  }
629  double exp1 = _formulae.power(eratio, 4.0);
630  hofe *= (1.0 - _formulae.expfun(-exp1));

633  if (x <= 4) {
