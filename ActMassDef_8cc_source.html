64 double ActMassDef::calcMass() {
65 
66  // a and z should be non-zero, so there should be no problems
67  // with dividing by zero etc...
68 
69  if (_a < 1e-6 || _z < 1e-6) {return 0.0;}
70 
71  double n = _a - _z;
72 
73  // First set xmass to Seeger formula without shell corrections
74  double xmass = 8.3674*n + 7.5848*_z - _alpha*_a;
75 
76  double oneThird = 1.0/3.0;
77  double twoThirds = 2.0*oneThird;
78  double zExp23 = pow(_z, twoThirds);
79  double aExp23 = pow(_a, twoThirds);
80  double aExp13 = pow(_a, oneThird);
81 
82  double term1 = 0.8076*_z*_z*(1.0 - (0.7636/zExp23) - (2.29/aExp23))/aExp13;
83  double term2 = _gamma*aExp23;
84  double a2z = _a - 2.0*_z;
85  double term3 = (_beta - (_eta/aExp13))*(a2z*a2z + 2.0*fabs(a2z))/_a;
86 
87  xmass += term1 + term2 + term3;
88 
89  // Now calculate j and k - these are the indices of the last filled
90  // neutron and proton shell, respectively
91 
92  int k(-1), j(-1), imagic(0);
93 
94  for (imagic = 0; imagic < _nmagicn; imagic++) {
95  if (_z < _magicn[imagic]) {break;}
96  k++;
97  }
98 
99  for (imagic = 0; imagic < _nmagicn; imagic++) {
100  if (n < _magicn[imagic]) {break;}
101  j++;
102  }
103 
104  // Now calculate normalised n and z: nprime, zprime
105  double nprime(1.0), zprime(1.0);
106 
107  if (k >= 0 && k < _nmagicn-1) {
108  double denom = _magicn[k+1] - _magicn[k];
109  if (fabs(denom) > 1e-30) {
110  zprime = (_z - _magicn[k])/denom;
111  }
112  }
113  if (j >= 0 && j < _nmagicn-1) {
114  double denom = _magicn[j+1] - _magicn[j];
115  if (fabs(denom) > 1e-30) {
116  nprime = (n - _magicn[j])/denom;
117  }
118  }
119 
120  double zarg = zprime*ActConstants::pi;
121  double narg = nprime*ActConstants::pi;
122  double sinZ = sin(zarg);
123  double sinN = sin(narg);
124  double sin2Z = sin(2.0*zarg);
125  double sin2N = sin(2.0*narg);
126 
127  // Now calculate s(jk)
128  double sjk = _ksi[j]*sinN + _ksi[k]*sinZ + _nu[j]*sin2N + _nu[k]*sin2Z;
129  sjk += (_phi[j] + _phi[k])*sinN*sinZ + _chi;
130 
131  // Add shell correction to xmass
132  xmass -= sjk;
133 
134  int ia = (int) _a;
135  int iz = (int) _z;
136  ActClassify classify(ia, iz);
137  int ioe = classify.classif();
138 
139  if (ioe == ActClassify::EvenEven) {
140  xmass -= del(_a);
141  } else if (ioe == ActClassify::OddOdd) {
142  xmass += del(_a);
143  }
144 
145  return xmass;
146 
147 }

157 double ActMassDef::thr(double zt, double at, double mdtgt, double mdprod) {
158 
159  // Calculate threshold energy for zt,at + p -> z,a + p + whatever, where
160  // whatever = max permitted alphas, at most one 3-H and at most one 3-He,
161  // plus required additional p,n.
162  // Also, assume incident particle remains after reaction, so things
163  // like (n,alpha) reactions are not handled. An (n,alpha) reaction
164  // would be calculated as (n,n+3-He).
165 
166  double thrse = 0.0;
167 
168  int nalpha(0), n3h(0), n3he(0);
169 
170  double defalp(2.42492), def3he(14.9309), def3h(14.9323);
171  double defn(8.071246), defp(7.288868);
172 
173  // Calculate changes of z and a in this reaction
174 
175  int dela = (int) (at - _a);
176  int delz = (int) (zt - _z);
177  int deln = dela - delz;
178 
179  if (delz >= 2 && deln >= 2) {nalpha = std::min(delz/2,deln/2);}
180 
181  // Are there enough knocked out nucleons left after forming alphas to
182  // make a 3-He too?
183 
184  int delns = deln - 2*nalpha;
185  int delzs = delz - 2*nalpha;
186  if (delzs >= 2 && delns >= 1) {
187  n3he = 1;
188  delzs -= 2;
189  delns--;
190  }
191 
192  // Are there still enough knocked out nucleons left to make a 3-H?
193 
194  if (delzs >= 1 && delns >= 2) {
195  n3h++;
196  delzs--;
197  delns -= 2;
198  }
199 
200  // Make up balance with free n and p
201  int nn = delns;
202  int np = delzs;
203 
204  thrse = mdprod + nalpha*defalp + def3he*n3he + def3h*n3h +
205  nn*defn + np*defp - mdtgt - defp;
206 
207  return thrse;
208
