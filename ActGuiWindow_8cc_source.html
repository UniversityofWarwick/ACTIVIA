794 void ActGuiWindow::saveInput(const QString& fileName) {
795 
796  // Save the various numbers and strings within the GUI into an output file
797  // that can be reloaded (avoiding the need to re-type the quantities).

808  saveData << nIsotopes << endl;

812 
813  QTableWidgetItem* AEntry = _targetTable->item(iIso, 0);

815  if (AEntry != 0 && fEntry != 0) {
816

821  saveData << "0.0 0.0" << endl;

823  }
824

835  saveData << isotopeFile.c_str() << endl;

837  if (_doSingle == true) {
838  int singleZ = this->getSingleZ();
839  double singleA = this->getSingleA();

865  double tdec = this->getDecayTime();

889 void ActGuiWindow::loadInput(const QString& fileName) {
890 
891  // Load the various numbers and strings within the GUI using the input file.
892  // The format of the input file follows that defined in the saveInput() function.
893 
894  std::string theFile = fileName.toStdString();
895  std::ifstream loadData(theFile.c_str());
896 
897  // Get the target definition
898  int targetZ(0);
899  loadData >> targetZ;
900  _targetZEdit->setText(QString::number(targetZ));
901 
902  int nIsotopes(0);
903  loadData >> nIsotopes;
904  this->updateTargetTable(nIsotopes);
905  _isotopeSpinBox->setValue(nIsotopes);
906 
907  int iIso;
908  double At(0.0), fraction(0.0);
909  for (iIso = 0; iIso < nIsotopes; iIso++) {
910 
911  loadData >> At >> fraction;
912 
913  QTableWidgetItem* AEntry = _targetTable->item(iIso, 0);
914  QTableWidgetItem* fEntry = _targetTable->item(iIso, 1);
915 
916  if (AEntry != 0) {
917  AEntry->setText(QString::number(At));
918  } else {
919  QTableWidgetItem* newAEntry = new QTableWidgetItem(QString::number(At));
920  _targetTable->setItem(iIso, 0, newAEntry);
921  }
922 
923  if (fEntry != 0) {
924  fEntry->setText(QString::number(fraction));
925  } else {
926  QTableWidgetItem* newfEntry = new QTableWidgetItem(QString::number(fraction));
927  _targetTable->setItem(iIso, 1, newfEntry);
928  }
929 
930  }
931 
932  // Specify the calculation mode
933  std::string calcType;
934  loadData >> calcType;
935  if (calcType == "z") {
936 
937  this->doSingle(true);
938  _ZButton->setChecked(true);
939 
940  } else {
941 
942  this->doAll(true);
943  _allButton->setChecked(true);
944 
945  }
946 
947  // Specify isotope data file
948  std::string isotopeDataFile("");
949  loadData >> isotopeDataFile;
950  if (_nuclideCombo != 0) {
951  QString isotopeString(isotopeDataFile.c_str());
952  _nuclideCombo->setItemText(0, isotopeString);
953  }
954 
955  // Specify single product Z,A (if needed)
956  if (_doSingle == true) {
957 
958  int prodZ(0);
959  double prodA(0.0);
960  loadData >> prodZ >> prodA;
961  _singleZEdit->setText(QString::number(prodA));
962  _singleAEdit->setText(QString::number(prodZ));
963  }
964 
965  // Specify beam energies
966  double EStart(0.0), EEnd(0.0), dE(0.0);
967  loadData >> EStart >> EEnd >> dE;
968  _minEEdit->setText(QString::number(EStart));
969  _maxEEdit->setText(QString::number(EEnd));
970  _dEEdit->setText(QString::number(dE));
971 
972  // Specify input data tables of cross-section data (if any)
973  std::string dataTablesFile("");
974  loadData >> dataTablesFile;
975 
976  if (dataTablesFile == "0") {
977 
978  _dataButton->setChecked(false);
979  _minSigmaEdit->setEnabled(false);
980  _minSigmaEdit->setReadOnly(true);
981  _dataTablesCombo->setEnabled(false);
982  _nuclideButton->setEnabled(false);
983 
984  } else {
985 
986  double minSigma(0.0);
987  loadData >> minSigma;
988 
989  _dataButton->setChecked(true);
990  _minSigmaEdit->setEnabled(true);
991  _minSigmaEdit->setReadOnly(false);
992  _minSigmaEdit->setText(QString::number(minSigma));
993  _dataTablesCombo->setEnabled(true);
994 
995  QString tableString(dataTablesFile.c_str());
996  _dataTablesCombo->setItemText(0, tableString);
997  _nuclideButton->setEnabled(true);
998 
999  }
1000 
1001  // Beam exposure and decay times
1002  double texp(0.0), tdec(0.0);
1003  loadData >> texp >> tdec;
1004  _expTimeEdit->setText(QString::number(texp));
1005  _decTimeEdit->setText(QString::number(tdec));
1006 
1007  // Finally, the output files, format and level of detail
1008  std::string xSecOutFile(""), decayOutFile("");
1009  int xSecFormat(0), xSecDetail(0);
1010  int decayFormat(0), decayDetail(0);
1011 
1012  loadData >> xSecOutFile >> xSecFormat >> xSecDetail;
1013  loadData >> decayOutFile >> decayFormat >> decayDetail;
1014 
1015  QString xSecOutString(xSecOutFile.c_str());
1016  _xSecOutBox->setItemText(0, xSecOutString);
1017  _outputSelection->setXSecFileName(xSecOutFile);
1018 
1019  QString decayOutString(decayOutFile.c_str());
1020  _decayOutBox->setItemText(0, decayOutString);
1021  _outputSelection->setDecayFileName(decayOutFile);
1022 
1023  QAbstractButton* xSecFormatButton = _xSecFormats->button(xSecFormat);
1024  if (xSecFormatButton != 0) {
1025  xSecFormatButton->setChecked(true);
1026  _outputSelection->setXSecType(xSecFormat);
1027  }
1028 
1029  QAbstractButton* decayFormatButton = _decayFormats->button(decayFormat);
1030  if (decayFormatButton != 0) {
1031  decayFormatButton->setChecked(true);
1032  _outputSelection->setDecayType(decayFormat);
1033  }
1034 
1035  QAbstractButton* xSecDetailButton = _xSecDetails->button(xSecDetail);
1036  if (xSecDetailButton != 0) {
1037  xSecDetailButton->setChecked(true);
1038  _outputSelection->setXSecDetail(xSecDetail);
1039  }
1040 
1041  QAbstractButton* decayDetailButton = _decayDetails->button(decayDetail);
1042  if (decayDetailButton != 0) {
1043  decayDetailButton->setChecked(true);
