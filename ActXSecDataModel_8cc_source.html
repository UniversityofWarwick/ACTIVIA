238  if (data != 0) {
239 
240  // First check if we already have the graph extracted in memory by checking if we
241  // have already processed the same target-product pair. This is
242  // significantly faster than trying to find the graph using map iterators.
243  int Zt = data->getizt();
244  int Z = data->getiz();
245  double At = data->getat();
246  double A = data->geta();
247 
248  if (Zt == _Zt && fabs(At - _At) < 1e-6 &&
249  Z == _Z && fabs(A - _A) < 1e-6) {
250 
251  // We have the same product and isotope information. Return the
252  // already found graph (or the null Graph - the first graph assigned).
253  return _theGraph;
254 
255  } else {
256

258  // Set the internal A and Z values for the target and product.

261  // information, then return the already found graph or the null graph.
262  _Zt = Zt; _At = At; _Z = Z; _A = A;
263  // Assign the graph to a null one at first.
264  _theGraph = _nullGraph;
265  // The remainder of this function will try to find the graph, if it exists.
266  }
267 
268  // Get the target nuclide pointer from the data
269  ActTargetNuclide* targetIsotope = data->getTargetIsotope();
270 
271  if (targetIsotope != 0) {
272
