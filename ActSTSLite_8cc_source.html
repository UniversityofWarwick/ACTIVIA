117  double sigma(0.0);
118

120  ActClassify classify(_ia, _iz);

126  double delac = 0.045*at36*(_logat + 1.23) + 31.5;
127

129  delac = 0.052*at36*(_loge - 3.17) + 31.5;
130  }
131  if (dela > delac) {dela = delac;}
132 
133  // Now get started with the cross-section calculations
134 
135  if (name == "lit1") {
136  this->calcLite1Values(ioe);
137  } else if (name == "lit2") {

139  } else if (name == "lit3") {
140  this->calcLite3Values(ioe);
141  } else if (name == "lit4") {
142  this->calcLite4Values(ioe);

145  return 0.0;
