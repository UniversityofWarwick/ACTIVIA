104 bool ActSTXSecAlgorithm::passSelection(ActNucleiData* data) {
105 
106  // Skip product nuclide if Z or A is too high.
107  // Ensure that we have at least two end products:
108  // X + beam -> Y_1 + Y_2 type reactions, not X + beam -> Y + gamma.
109  double ztgt = data->getzt();

111  double zCut = ztgt + zBeam;
112 
113  double atgt = data->getat();
114  double aBeam = data->getabeam();
115  double aCut = atgt + aBeam - 1.0;
116 
117  bool passed(true);

120  double a = data->geta();

122
